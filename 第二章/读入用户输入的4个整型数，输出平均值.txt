#include <iostream>
#include <iomanip>


using namespace std;

int main()
{
    const int total = 4;
    double num[total], sum = 0, average;
    for (int i=0; i < total; i += 1) {
        cin >> num[i];
        sum += num[i];
    }
    cout << num[0] << "、" << num[1] << "、" << num[2] << "、" << num[3] ;
    average = sum / total;
    cout << setiosflags(ios::fixed) << setprecision(2) << "的平均值是" << average;
}
